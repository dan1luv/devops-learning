=== Урок 6 - Docker ===
docker run IMAGE - запустить контейнер
docker ps -a - все контейнеры, включая остановленные
docker images - список образов
docker build -t NAME -f Dockerfile . - собрать образ
docker run -it IMAGE bash - войти внутрь контейнера
docker run -v  /host/path:/container/path - примонтировать папку
FROM - базовый образ
RUN - команда при сборке
COPY - скопировать файл в образ
WORKDIR - рабочая папка внутри контейнера
CMD - команда при запуске контейнера
Слои кешируются - изменил строку = пересобрались только слои ниже

=== Урок 7 - Docker Compose ===
docker compose up - запустить стек
docker compose up -d - запустить в фоне
docker compose down - остановить и удалить контейнеры + сеть 
docker compose ps - статус контейнеров
docker compose logs SERVICE - логи сервиса
depends_on - порядок запуска сервисов
services - список сервисов в docker-compose.yml
ports: "8080:80" - проброс портов хост:контейнер
Контейнеры общаются через сеть по имени сервиса

=== Урок 8 - GitHub Actions ===
.github/workflows/*.yml - файлы пайплайнов
on: push - триггер при git push
on: schedule: cron - триггер по расписанию
jobs - список задач
runs-on: ubuntu-latest - чистая Ubuntu на серверах GitHub
steps - шаги внутри задачи
uses: actions/checkout@v4 - готовое действие
run: - выполнить shell команду
каждый push = запуск пайплайна
